    <script>
        const html5QrCode = new Html5Qrcode("reader");
        const qrConfig = { fps: 10, qrbox: { width: 250, height: 250 } };

        html5QrCode.start(
            { facingMode: "environment" }, 
            qrConfig,
            (decodedText) => {
                document.getElementById('scan-id').value = decodedText;
                navigator.vibrate(50);
            }
        ).catch(err => console.error("相機啟動失敗", err));

        async function uploadData() {
            const btn = document.getElementById('submit-btn');
            const status = document.getElementById('status');
            
            // 重要：請將下方的網址替換成您剛才部署得到的 Google Apps Script URL
            const scriptUrl = 'https://script.google.com/macros/s/AKfycbyEMr0PyC_sr1tmJUYBfqyadDnsmxv62LR4OLerm8o1jZuSlZI8FT3nvxLJg-cYCNJCJw/exec'; 

            const payload = {
                id: document.getElementById('scan-id').value,
                info: document.getElementById('cust-info').value,
                notes: document.getElementById('notes').value,
                time: new Date().toLocaleString()
            };

            if (!payload.id && !payload.info) {
                alert("請至少掃碼或輸入客戶資訊");
                return;
            }

            btn.disabled = true;
            status.innerText = "上傳中...";

            try {
                const response = await fetch(scriptUrl, {
                    method: 'POST',
                    mode: 'no-cors',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });

                status.style.color = "#4caf50";
                status.innerText = "儲存成功！已更新至雲端試算表。";
                
                document.getElementById('scan-id').value = "";
                document.getElementById('cust-info').value = "";
                document.getElementById('notes').value = "";
                document.getElementById('business-card').value = "";

            } catch (error) {
                status.style.color = "#ff5252";
                status.innerText = "上傳失敗，請檢查網路。";
            } finally {
                btn.disabled = false;
            }
        }
    </script>
